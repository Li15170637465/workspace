#include "music_player.h"
#include <windows.h>

int main() {
    PlayerState* player = init_player();
    play_music(player, "D:\\workspace\\AI_Make\\music.mp3"); // 替换为你的实际文件路径
    
    Sleep(3000);
    pause_music(player);
    Sleep(1000);
    resume_music(player);
    Sleep(2000);
    stop_music(player);
    
    cleanup_player(player);
    return 0;
}
